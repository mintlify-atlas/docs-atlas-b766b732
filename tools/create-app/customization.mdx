# Customization

Learn how to customize your generated MCP application to fit your needs.

## Server Configuration

Customize your MCP server in `src/index.ts`:

### Basic Settings

```typescript src/index.ts
import { MCPServer } from "mcp-use/server";

const server = new MCPServer({
  // Required
  name: "my-server",
  version: "1.0.0",

  // Display Information
  title: "My MCP Server",              // Display name
  description: "My awesome MCP server",  // Description

  // URLs
  baseUrl: "http://localhost:3000",     // Full base URL
  websiteUrl: "https://mycompany.com",  // Company website

  // Branding
  favicon: "favicon.ico",               // Favicon path
  icons: [
    {
      src: "icon.svg",
      mimeType: "image/svg+xml",
      sizes: ["512x512"],
    },
  ],
});
```

### Environment Variables

Use environment variables for configuration:

```typescript src/index.ts
const server = new MCPServer({
  name: process.env.SERVER_NAME || "my-server",
  version: process.env.SERVER_VERSION || "1.0.0",
  baseUrl: process.env.MCP_URL || "http://localhost:3000",
});

const PORT = process.env.PORT ? parseInt(process.env.PORT) : 3000;
server.listen(PORT);
```

Create `.env` file:

```bash .env
SERVER_NAME=my-production-server
SERVER_VERSION=2.1.0
MCP_URL=https://api.mycompany.com
PORT=8080

# API Keys
OPENAI_API_KEY=sk-...
DATABASE_URL=postgres://...
```

<Warning>
  Never commit `.env` to Git! It's already in `.gitignore`.
</Warning>

### Example: Modular Tools

```typescript src/tools/weather.ts
import { z } from "zod";
import { text } from "mcp-use/server";

export const weatherTool = {
  name: "fetch-weather",
  description: "Fetch weather for a city",
  schema: z.object({
    city: z.string(),
    units: z.enum(["celsius", "fahrenheit"]).default("celsius"),
  }),
  handler: async ({ city, units }) => {
    // Implementation
    return text(`Weather in ${city}`);
  },
};
```

```typescript src/tools/index.ts
export { weatherTool } from "./weather";
export { databaseTool } from "./database";
```

```typescript src/index.ts
import { MCPServer } from "mcp-use/server";
import * as tools from "./tools";

const server = new MCPServer({ /* ... */ });

// Register all tools
for (const tool of Object.values(tools)) {
  server.tool(tool, tool.handler);
}

server.listen(3000);
```

Customize React widgets in the `resources/` directory:

### Styling Options

<Tabs>
  <Tab title="Tailwind CSS">

### Widget Configuration

Customize widget metadata:

```tsx resources/my-widget.tsx
import { z } from "zod";
import type { WidgetMetadata } from "mcp-use/react";

const propSchema = z.object({
  title: z.string().describe("Widget title"),
  data: z.array(z.number()).describe("Chart data"),
  theme: z.enum(["light", "dark"]).default("light"),
});

export const widgetMetadata: WidgetMetadata = {
  title: "My Custom Widget",
  description: "A widget that displays custom data",
  props: propSchema,
};

export default function MyWidget() {
  const { props } = useMcp<z.infer<typeof propSchema>>();
  // Use props.title, props.data, props.theme
}
```

### Example: Using External API

```typescript src/tools/weather.ts
import axios from "axios";

server.tool(
  {
    name: "fetch-weather",
    schema: z.object({ city: z.string() }),
  },
  async ({ city }) => {
    const response = await axios.get(
      `https://api.weatherapi.com/v1/current.json`,
      {
        params: {
          key: process.env.WEATHER_API_KEY,
          q: city,
        },
      }
    );

    return object(response.data);
  }
);
```

### Example: Using Date Library

}

### Example Tests

```typescript tests/tools.test.ts
import { describe, it, expect } from "vitest";
import { weatherTool } from "../src/tools/weather";

describe("Weather Tool", () => {
  it("should fetch weather for a city", async () => {
    const result = await weatherTool.handler({
      city: "San Francisco",
      units: "celsius",
    });

    expect(result.content[0].type).toBe("text");
    expect(result.content[0].text).toContain("San Francisco");
  });

  it("should validate schema", () => {
    expect(() => {
      weatherTool.schema.parse({ city: "NYC" });
    }).not.toThrow();

    expect(() => {
      weatherTool.schema.parse({ city: 123 });
    }).toThrow();
  });
});
```
